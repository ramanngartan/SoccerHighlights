git init --initializes the git directory

git add (name of file) --adds file to the stage area

git commit -m (msg that goes with the file!) --adds file to local repo..

git remote add origin (link of github repo..) --adds file to remote repo on github 
//here "origin" can be replaced with anything and "master" is a branch

git push -u origin master --pushes file to remote repo

gitignore --make this file in the local repo and keep all the secret files inside it

git clone (link from github) --sets up all the files locally

git branch (name of branch) --creates another branch

git branch --shows the names of branches

git checkout (name of branch to go to) --takes you to nother branch

git status --shows whats going on in git

git log --tells the recent changes made in git
